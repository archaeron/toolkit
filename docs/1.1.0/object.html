Object

  • array Values (object)

    Gets all the enumerable values within an object.

    View Example{% highlight js %} Object.values({1:2,3:4}); // returns [2,4] {% endhighlight %}
    View Code{% highlight js %}function (obj) { var arr = []; for (key in obj) { arr.push(obj[key]); } return arr; }{% endhighlight %}

  • void ForEach (object)

    A fast looping mechanism for objects. Like Array.forEach.

    View Example{% highlight js %} Object.forEach({1:2,3:4}, function(key, val) { log(key + ':' + val) }); // > 1:2 // > 3:4 {% endhighlight %}
    View Code{% highlight js %}function (obj, callback) { return Object.keys(obj).forEach(function(key) { return callback.call(null, key, obj[key], obj); }); }{% endhighlight %}

  • bool IsObject (*objects)

    Returns whether all the given parameters are objects

    View Example{% highlight js %} Object.isObject({1:2,3:4}); // returns true Object.isObject(function() { }); // returns false Object.isObject({1:2,3:4}, function() { }); // returns false {% endhighlight %}
    View Code{% highlight js %}function () { return Object.values(arguments).every(function(value) { return typeof value === 'object'; }); }{% endhighlight %}

  • mixed Each (object, function)

    Provides a utility to quickly iterate through enumerable properties of an object.

    View Example{% highlight js %} var ret = Object.each({1:2,3:4}, function(key, val) { return (key + ':' + val); }); // ret = 1:2 {% endhighlight %}
    View Code{% highlight js %}function (obj, callback) { var key, result; for (key in obj) if ((result = callback.call(null, key, obj[key], this)) !== undefined) return result; }{% endhighlight %}

  • object Map (object, function)

    Similar to Array.map except for enumerable object properties.

    View Example{% highlight js %} Object.map({1:2,3:4}, function(key, val) { return key * val; }); // returns {1:2,3:12} {% endhighlight %}
    View Code{% highlight js %}function (object, callback) { var obj = Object.clone(object); Object.walk(obj, callback); return obj; }{% endhighlight %}

  • array GetOwnPropertyDescriptors (object)

    Retrieves object property descriptors for every property within an object

    View Example{% highlight js %} Object.getOwnPropertyDescriptors({1:2,3:4}); // returns { // '1': { // value: 2, // writable: true, // enumerable: true, // configurable: true // }, // '3': { // value: 4, // writable: true, // enumerable: true, // configurable: true // } // } {% endhighlight %}
    View Code{% highlight js %}function (object) { var descriptors = {}; Object.getOwnPropertyNames(object).forEach(function(key) { descriptors[key] = Object.getOwnPropertyDescriptor(object, key); }); return descriptors; }{% endhighlight %}

  • self Walk (object, function)

    Similar to Object.map except that the callback values are applied to the object instance and not a clone.

    View Example{% highlight js %} var obj = {1:2,3:4}; Object.map(obj, function(key, val) { return key * val; }); // obj = {1:2,3:12} {% endhighlight %}
    View Code{% highlight js %}function (obj, callback) { Object.forEach(obj, function(key, val) { obj[key] = callback.call(null, key, val); }); return obj; }{% endhighlight %}

  • mixed Reduce (object, callback, [start=undefined])

    Like Array.reduce except for objects. Reduces the object down into a single value.

    View Example{% highlight js %} Object.reduce({1:2,3:4}, function(group, key, val) { return group + key + val; }, 0); // returns 10 (0+1+2+3+4) {% endhighlight %}
    View Code{% highlight js %}function (object, callback, start) { Object.forEach(obj, function(key, val) { start = callback.call(null, start, key, obj[key]); }); return start; }{% endhighlight %}

  • object Merge (object, [level=0])

    Like Array.reduce except for objects. Reduces the object down into a single value. You can pass the first parameter as an array or treat it as *objects and ignore the level.

    View Example{% highlight js %} Object.merge({1:2}, {3:4}, {5:6}); // returns {1:2,3:4,5:6} Object.merge([{1:2}, {3:4}, {5:6}]); // returns {1:2,3:4,5:6} Object.merge({1:2,3:{4:5}}, {1:3,3:{5:6}}); // returns {1:3,3:{5:6}} // The second parameter with the first being an array of objects is how many levels deep to merge (recursive merge). This defaults to 0. -1 will recursively merge indefinitely. Object.merge([{1:2,3:{4:5}}, {1:3,3:{5:6}}], 1); // returns {1:3,3:{4:5,5:6}} {% endhighlight %}
    View Code{% highlight js %}function (objects, level) { if (Object.isObject(objects, level)) { objects = Object.values(arguments), level = 0; } level = (level === undefined) ? 0 : level; return objects.reduce(function(group, obj) { Object.forEach(obj, function(key, val) { if ( ! val in group || level == 0 || ! Object.isObject(group[key]) || ! Object.isObject(val)) { group[key] = val; } else { group[key] = Object.merge([group[key], val], level - 1); } }); return group; }, {}); }{% endhighlight %}

  • object Clone (object, [inherit=true])

    Like Array.reduce except for objects. Reduces the object down into a single value.

    View Example{% highlight js %} function A() { } Object.defineProperty(A.prototype, 'foo', { value: 'bar' }); var a = new A(), b = Object.clone(a); a.foo == b.foo // returns true a == b // returns false {% endhighlight %}
    View Code{% highlight js %}function (obj, inherit) { inherit = (inherit === undefined) ? true : false; return Object.create(Object.getPrototypeOf(obj), inherit ? Object.getOwnPropertyDescriptors(obj) : undefined); }{% endhighlight %}

  • object Filter (object, callback)

    Like Array.filter except for objects. Only enumerable values are filtered.

    View Example{% highlight js %} Object.filter({1:2,3:4,5:6}, function(key, val) { return key == 3; }); // returns {3:4} {% endhighlight %}
    View Code{% highlight js %}function (object, callback) { var obj = Object.clone(object); Object.each(object, function(key, val) { if (callback.call(null, key, val) === false) { delete obj[key]; } }); return obj; }{% endhighlight %}

  • object Clean (object)

    Like Array.clean except for objects. The following values are filtered: NaN, undefined, null, 0 or false

    View Example{% highlight js %} Object.clean({1:false,2:0,3:NaN,4:null,5:6}); // returns {5:6} {% endhighlight %}
    View Code{% highlight js %}function (obj) { return Object.filter(obj, function(key, val) { return !! val; }); }{% endhighlight %}
