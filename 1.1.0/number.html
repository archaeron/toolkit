Number.prototype

  • int Gcd (*int)

    Calculates the Greatest common divisor between a set of numbers. Also known as the greatest common factor. This uses Stein's binary algorithm.

    View Example{% highlight js %} (12).gcd(6, 9) // returns 3 {% endhighlight %}
    View Code{% highlight js %}function () { return Object.values(arguments).reduce(function(a, b) { if (a == 0 || b == 0) return a | b; for (var shift = 0; ((a | b) & 1) == 0; shift++) { a >>= 1; b >>= 1; } while ((a & 1) == 0) a >>= 1; do { while ((b & 1) == 0) b >>= 1; if (a < b) b -= a; else { var diff = a - b; a = b; b = diff; } b >>= 1; } while (b != 0); return a << shift; }, this); }{% endhighlight %}

  • int Lcm (*int)

    Calculates the lowest common multiple between a set of numbers. Uses GCD within the caluclation.

    View Example{% highlight js %} (21).gcd(6) // returns 42 {% endhighlight %}
    View Code{% highlight js %}function () { var nums = Object.values(arguments), gcd = this.gcd.apply(this, nums); return Math.abs(nums.product() * this) / gcd; }{% endhighlight %}
